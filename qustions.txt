Q1)Inverse star pattern

Q2)accept 10 numbers and display max ,mean and averge value  

Ans q2
max=0
min=0
sum=0
for((i=0;i<10;i++))
do

read -p "Enter num" num
sum=$((sum+num))
if [ $i -eq 1 ]; then
min=$num
max=$num
else
if [ $num -lt $min ]; then
min=$num
fi
if [ $num -gt $max ]; then
max=$num
fi
fi
done
avg=$((sum/10))
echo $min
echo $max
echo $avg
